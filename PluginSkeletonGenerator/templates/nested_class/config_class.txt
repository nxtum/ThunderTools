class Config : public Core::JSON::Container {
private:
~INDENT_INCREASE~
    Config(const Config&) = delete;
    Config& operator=(const Config&) = delete;
~INDENT_DECREASE~
public:
~INDENT_INCREASE~
    Config() 
~INDENT_INCREASE~
        : Example("Example")
~INDENT_DECREASE~
    {
~INDENT_INCREASE~
        Add(_T("example"), &Example);
~INDENT_DECREASE~
    }
    ~Config(){}
~INDENT_DECREASE~
public:
~INDENT_INCREASE~
Core::JSON::String Example;
~INDENT_DECREASE~
}
